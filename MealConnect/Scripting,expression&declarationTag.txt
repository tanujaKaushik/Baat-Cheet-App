Create a JSP page and implement a Scripting Tag, Expression tag and 
Declaration tag. 

<%@ page language="java" contentType="text/html; charset=UTF-8"
    pageEncoding="UTF-8"%>
<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <title>JSP Tags Example</title>
</head>
<body>
    <h2>JSP Tags Example</h2>

    <!-- Scriptlet Tag -->
    <% 
        // Scriptlet tag - Java code goes here
        String message = "Hello, JSP!";
        out.println("<p>Message from Scriptlet Tag: " + message + "</p>");
    %>

    <!-- Expression Tag -->
    <p>Message from Expression Tag: <%= "Hello, JSP!" %></p>

    <!-- Declaration Tag -->
    <%! 
        // Declaration tag - Variable declaration and method definition go here
        private int calculateSquare(int num) {
            return num * num;
        }
    %>

    <p>Example using Declaration Tag: The square of 5 is <%= calculateSquare(5) %></p>

</body>
</html>
