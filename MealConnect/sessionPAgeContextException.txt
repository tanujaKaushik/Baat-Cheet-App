Write a JSP program using the following implicit objects with an example: 
a. Out 
b. request 
c. response 
d. Session 
e. pageContext 
f. exception 


<%@ page language="java" contentType="text/html; charset=UTF-8"
    pageEncoding="UTF-8"%>
<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <title>Implicit Objects Example</title>
</head>
<body>
    <h2>Implicit Objects Example</h2>

    <%-- a. out (JspWriter) --%>
    <%
        out.println("<p><b>a. Out (JspWriter)</b></p>");
        out.println("<p>Hello from Out implicit object!</p>");
    %>

    <%-- b. request (HttpServletRequest) --%>
    <%
        out.println("<p><b>b. request (HttpServletRequest)</b></p>");
        String userAgent = request.getHeader("User-Agent");
        out.println("<p>User Agent: " + userAgent + "</p>");
    %>

    <%-- c. response (HttpServletResponse) --%>
    <%
        out.println("<p><b>c. response (HttpServletResponse)</b></p>");
        response.setContentType("text/html");
        response.getWriter().println("<p>Hello from Response implicit object!</p>");
    %>

    <%-- d. session (HttpSession) --%>
    <%
        out.println("<p><b>d. session (HttpSession)</b></p>");
        session.setAttribute("username", "JohnDoe");
        String username = (String) session.getAttribute("username");
        out.println("<p>Welcome, " + username + "!</p>");
    %>

    <%-- e. pageContext (PageContext) --%>
    <%
        out.println("<p><b>e. pageContext (PageContext)</b></p>");
        pageContext.setAttribute("message", "This is a message from PageContext");
        String message = (String) pageContext.getAttribute("message");
        out.println("<p>Message: " + message + "</p>");
    %>

    <%-- f. exception (Throwable) --%>
    <%
        out.println("<p><b>f. exception (Throwable)</b></p>");
        try {
            int num = 10 / 0; // Causes ArithmeticException
        } catch (ArithmeticException e) {
            out.println("<p>ArithmeticException caught: " + e.getMessage() + "</p>");
        }
    %>

</body>
</html>




or 





<%@ page language="java" contentType="text/html; charset=UTF-8"
    pageEncoding="UTF-8" %>

<!DOCTYPE html>
<html>
<head>
<title>JSP Implicit Objects Example</title>
</head>
<body>

<%-- 1. out Object --%>
<h2>Hello! This is some output using the out object.</h2>

<%-- 2. request Object --%>
<p>Request Parameter (if any): 
  <% String name = request.getParameter("name"); 
     if (name != null) {
       out.println(name);
     } else {
       out.println("No name parameter found.");
     }
  %>
</p>

<%-- 3. response Object --%>
<%-- (demonstrate setting a header) --%>
<% response.setHeader("Content-Type", "text/plain"); %>

<%-- 4. Session Object --%>
<%-- (assuming a session attribute is set) --%>
<% 
  HttpSession session = request.getSession(false);  // Don't create if not existing
  if (session != null && session.getAttribute("message") != null) {
    out.println("<p>Session Message: " + session.getAttribute("message") + "</p>");
  }
%>

<%-- 5. pageContext Object --%>
<%-- (demonstrate accessing a servlet context attribute) --%>
<% 
  String appVersion = (String) pageContext.getServletContext().getAttribute("appVersion");
  if (appVersion != null) {
    out.println("<p>Application Version: " + appVersion + "</p>");
  }
%>

<%-- 6. exception Object --%>
<%-- (simulate an exception and access its message) --%>
<% 
  try {
    int result = 10 / 0;  // This will cause an exception
  } catch (ArithmeticException e) {
    out.println("<p>Error: " + e.getMessage() + "</p>");
  }
%>

</body>
</html>
